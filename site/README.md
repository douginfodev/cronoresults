## CONVERTER DP IN PIXEL
MDPI = 1DP = 1PX
HDPI = 1dp =  1.5px
XHDPI = 1dp = 2px
XXHDPI = 1dp = 3px
XXX HDPI = 1dp = 4px

formula mdpi = px = 1dp * (160 dpi / 160)
formula hdpi = px = 1dp * (240 dpi / 160)

formula xxxdpi = px = 1dp * (640 dpi / 160)
